const KEYWORD_MAP = {

};

function parseSummary(text) {
  const lines = text.split("\n").map(l => l.trim()).filter(Boolean);
  const result = { patient: [], diagnosis: [], medications: [], instructions: [], followup: [], warnings: [] };
  lines.forEach(line => {
    const lower = line.toLowerCase();
    let placed = false;
    for (const [section, keywords] of Object.entries(KEYWORD_MAP)) {
      if (keywords.some(k => lower.includes(k))) {
        result[section].push(line);
        placed = true;
        break;
      }
    }
    if (!placed) result.instructions.push(line);
  });
  for (const key of Object.keys(result)) result[key] = [...new Set(result[key])];
  return result;
}

function cleanLine(line) { return line.replace(/^[-•*]\s*/, "").trim(); }

function speak(text, lang = "en") {
  if (!window.speechSynthesis) return;
  window.speechSynthesis.cancel();
  const u = new SpeechSynthesisUtterance(text);
  u.lang = lang === "Hindi" ? "hi-IN" : "en-US";
  u.rate = 0.9;
  window.speechSynthesis.speak(u);
}
function stopSpeaking() { if (window.speechSynthesis) window.speechSynthesis.cancel(); }

// ── Claude API call (text) ────────────────────────────────────────
async function callClaude(system, userText) {
  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-20250514",
      max_tokens: 1000,
      system,
      messages: [{ role: "user", content: userText }],
    }),
  });
  const data = await res.json();
  return data.content?.[0]?.text || "";
}

// ── Claude Vision: extract text from image ────────────────────────
async function extractTextFromImage(base64, mediaType) {
  const res = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      model: "claude-sonnet-4-20250514",
      max_tokens: 1000,
      system: "You are a medical document OCR assistant. Extract ALL text from the image exactly as it appears. Preserve line breaks and formatting. Output only the extracted text, nothing else.",
      messages: [{
        role: "user",
        content: [
          { type: "image", source: { type: "base64", media_type: mediaType, data: base64 } },
          { type: "text", text: "Extract all text from this hospital discharge summary document." }
        ]
      }],
    }),
  });
  const data = await res.json();
  return data.content?.[0]?.text || "";
}

async function explainTerm(term, language) {
  const langNote = language === "Hindi"
    ? "Respond entirely in simple Hindi (Devanagari). Use everyday Hindi words."
    : "Respond in simple English a non-medical person can understand.";
  return callClaude(
    `You are a friendly medical explainer. ${langNote} Explain in 3-4 simple sentences. Use an analogy if helpful. Be warm. No jargon.`,
    `Explain this medical term simply: "${term}"`
  );
}

          {inputMode === "image" && (
            <div>
              <h2 style={{ margin: "0 0 4px", fontSize: 14, fontWeight: 700, color: "#1e40af", textTransform: "uppercase", letterSpacing: "0.04em" }}>Upload Discharge Summary Image</h2>
              <p style={{ margin: "0 0 16px", fontSize: 12, color: "#94a3b8" }}>JPG, PNG — AI will read the text automatically using vision</p>

              {/* Drop Zone */}
              {!imgPreview ? (
                <div>
                  {/* Two buttons: Gallery + Camera */}
                  <div style={{ display: "flex", gap: 12, marginBottom: 14, flexWrap: "wrap" }}>
                    {/* Gallery/File picker */}
                    <label style={{
                      flex: 1, minWidth: 140, display: "flex", flexDirection: "column", alignItems: "center", justifyContent: "center",
                      gap: 8, padding: "28px 16px", border: "2px dashed #bfdbfe", borderRadius: 14,
                      cursor: "pointer", background: "#eff6ff", transition: "all 0.2s", textAlign: "center",
                    }}
                      onMouseEnter={e => e.currentTarget.style.background = "#dbeafe"}
                      onMouseLeave={e => e.currentTarget.style.background = "#eff6ff"}>
                      <span style={{ fontSize: 36 }}>🖼️</span>
                      <span style={{ fontSize: 14, fontWeight: 700, color: "#1e40af" }}>Gallery se Upload</span>
                      <span style={{ fontSize: 12, color: "#94a3b8" }}>JPG, PNG select karo</span>
                      <input type="file" accept="image/*" onChange={handleImageSelect} style={{ display: "none" }} />
                    </label>

                    {/* Camera */}
                    <label style={{
                      flex: 1, minWidth: 140, display: "flex", flexDirection: "column", alignItems: "center", justifyContent: "center",
                      gap: 8, padding: "28px 16px", border: "2px dashed #bbf7d0", borderRadius: 14,
                      cursor: "pointer", background: "#f0fdf4", transition: "all 0.2s", textAlign: "center",
                    }}
                      onMouseEnter={e => e.currentTarget.style.background = "#dcfce7"}
                      onMouseLeave={e => e.currentTarget.style.background = "#f0fdf4"}>
                      <span style={{ fontSize: 36 }}>📸</span>
                      <span style={{ fontSize: 14, fontWeight: 700, color: "#059669" }}>Camera se Photo Lo</span>
                      <span style={{ fontSize: 12, color: "#94a3b8" }}>Live camera open hoga</span>
                      <input type="file" accept="image/*" capture="environment" onChange={handleImageSelect} style={{ display: "none" }} />
                    </label>
                  </div>

                  {/* Drag drop zone */}
                  <div
                    onDragOver={e => { e.preventDefault(); e.currentTarget.style.borderColor = "#3b82f6"; e.currentTarget.style.background = "#eff6ff"; }}
                    onDragLeave={e => { e.currentTarget.style.borderColor = "#e2e8f0"; e.currentTarget.style.background = "#f8fafc"; }}
                    onDrop={e => {
                      e.preventDefault();
                      e.currentTarget.style.borderColor = "#e2e8f0";
                      e.currentTarget.style.background = "#f8fafc";
                      const file = e.dataTransfer.files?.[0];
                      if (file) handleImageSelect({ target: { files: [file] } });
                    }}
                    style={{ border: "2px dashed #e2e8f0", borderRadius: 12, padding: "16px", textAlign: "center", background: "#f8fafc", transition: "all 0.2s" }}>
                    <p style={{ margin: 0, fontSize: 13, color: "#94a3b8" }}>📂 Ya yahan drag & drop karo</p>
                  </div>
                </div>
              ) : (
                <div>
                  {/* Image Preview */}
                  <div style={{ position: "relative", marginBottom: 14 }}>
                    <img src={imgPreview} alt="Uploaded document" style={{ width: "100%", maxHeight: 280, objectFit: "contain", borderRadius: 12, border: "1.5px solid #e2e8f0", background: "#f8fafc" }} />
                    <button onClick={clearImage} style={{ position: "absolute", top: 8, right: 8, background: "rgba(0,0,0,0.55)", border: "none", color: "#fff", borderRadius: 8, padding: "4px 10px", fontSize: 12, cursor: "pointer", fontFamily: "inherit", fontWeight: 600 }}>✕ Remove</button>
                  </div>

                  {ocrError && <div style={{ marginBottom: 12, padding: "9px 14px", background: "#fef2f2", border: "1px solid #fecaca", borderRadius: 8, color: "#dc2626", fontSize: 13 }}>⚠️ {ocrError}</div>}

                  <div style={{ display: "flex", gap: 10, alignItems: "center", flexWrap: "wrap" }}>
                    <label style={{ padding: "8px 16px", background: "#f1f5f9", border: "1px solid #e2e8f0", color: "#475569", borderRadius: 10, cursor: "pointer", fontSize: 13, fontFamily: "inherit", fontWeight: 600 }}>
                      🖼️ Gallery
                      <input type="file" accept="image/*" onChange={handleImageSelect} style={{ display: "none" }} />
                    </label>
                    <label style={{ padding: "8px 16px", background: "#f0fdf4", border: "1px solid #bbf7d0", color: "#059669", borderRadius: 10, cursor: "pointer", fontSize: 13, fontFamily: "inherit", fontWeight: 600 }}>
                      📸 Camera
                      <input type="file" accept="image/*" capture="environment" onChange={handleImageSelect} style={{ display: "none" }} />
                    </label>
                    <button onClick={handleOCR} disabled={ocrLoading} style={{
                      marginLeft: "auto", padding: "9px 24px",
                      background: ocrLoading ? "#94a3b8" : "linear-gradient(135deg,#2563eb,#0891b2)",
                      border: "none", borderRadius: 10, color: "#fff", fontSize: 14,
                      cursor: ocrLoading ? "not-allowed" : "pointer", fontFamily: "inherit", fontWeight: 700,
                      boxShadow: ocrLoading ? "none" : "0 4px 14px rgba(37,99,235,0.35)",
                      display: "flex", alignItems: "center", gap: 8,
                    }}>
                      {ocrLoading ? (
                        <><span style={{ display: "inline-block", animation: "spin 1s linear infinite" }}>⟳</span> Reading Image...</>
                      ) : "🔍 Read & Extract Text"}
                    </button>
                  </div>

                  {ocrLoading && (
                    <div style={{ marginTop: 14, padding: "12px 16px", background: "#eff6ff", border: "1px solid #bfdbfe", borderRadius: 10, fontSize: 13, color: "#1e40af", display: "flex", alignItems: "center", gap: 10 }}>
                      <span style={{ fontSize: 20, display: "inline-block", animation: "spin 1s linear infinite" }}>⟳</span>
                      AI image padh raha hai... thoda wait karo 🏥
                    </div>
                  )}
                </div>
              )}
            </div>
          )}
        </div>

        {/* Extracted text preview (after OCR) */}
        {rawText && inputMode === "image" && (
          <div style={{ background: "#fff", borderRadius: 16, boxShadow: "0 4px 16px rgba(0,0,0,0.07)", padding: 22, marginBottom: 22, border: "1px solid #bbf7d0" }}>
            <div style={{ display: "flex", alignItems: "center", justifyContent: "space-between", marginBottom: 12, flexWrap: "wrap", gap: 8 }}>
              <div>
                <h3 style={{ margin: 0, fontSize: 14, fontWeight: 700, color: "#059669", textTransform: "uppercase", letterSpacing: "0.04em" }}>✅ Text Extracted from Image</h3>
                <p style={{ margin: "2px 0 0", fontSize: 12, color: "#94a3b8" }}>Review karo phir Parse karo</p>
              </div>
              <button onClick={handleParse} style={{ padding: "8px 22px", background: "linear-gradient(135deg,#059669,#0891b2)", border: "none", borderRadius: 10, color: "#fff", fontSize: 13, cursor: "pointer", fontFamily: "inherit", fontWeight: 700, boxShadow: "0 4px 12px rgba(5,150,105,0.35)" }}>
                🔍 Parse & Simplify
              </button>
            </div>
            <textarea value={rawText} onChange={e => setRawText(e.target.value)}
              style={{ width: "100%", height: 160, background: "#f0fdf4", border: "1.5px solid #bbf7d0", borderRadius: 10, color: "#334155", padding: "12px 14px", fontSize: 13, fontFamily: "monospace", resize: "vertical", outline: "none", lineHeight: 1.6, boxSizing: "border-box" }} />
          </div>
        )}

        {/* Results */}
        {results && (
          <div style={{ background: "#fff", borderRadius: 18, boxShadow: "0 4px 24px rgba(0,0,0,0.08)", border: "1px solid #e2e8f0", overflow: "hidden" }}>
            <div style={{ padding: "13px 20px", background: "#f8fafc", borderBottom: "1px solid #e2e8f0", display: "flex", gap: 10, alignItems: "center", flexWrap: "wrap" }}>
              <input value={search} onChange={e => setSearch(e.target.value)} placeholder="🔎 Search within results..."
                style={{ flex: 1, minWidth: 160, padding: "7px 13px", border: "1.5px solid #e2e8f0", borderRadius: 8, fontSize: 13, fontFamily: "inherit", outline: "none", background: "#fff", color: "#334155" }} />
              <button onClick={copyAll} style={{ padding: "7px 14px", background: copied ? "#d1fae5" : "#f0fdf4", border: `1px solid ${copied ? "#6ee7b7" : "#bbf7d0"}`, color: copied ? "#059669" : "#16a34a", borderRadius: 8, cursor: "pointer", fontSize: 13, fontFamily: "inherit", fontWeight: 600 }}>
                {copied ? "✓ Copied!" : "📋 Copy All"}
              </button>
              <button onClick={handlePrint} style={{ padding: "7px 14px", background: "#eff6ff", border: "1px solid #bfdbfe", color: "#2563eb", borderRadius: 8, cursor: "pointer", fontSize: 13, fontFamily: "inherit", fontWeight: 600 }}>🖨️ Print</button>
            </div>

            <div style={{ display: "flex", overflowX: "auto", borderBottom: "1px solid #e2e8f0" }}>
              {SECTIONS.map(s => (
                <button key={s.key} onClick={() => { setActiveTab(s.key); setSearch(""); }} style={{
                  padding: "12px 17px", border: "none", borderBottom: activeTab === s.key ? `3px solid ${s.color}` : "3px solid transparent",
                  background: activeTab === s.key ? `${s.color}11` : "transparent",
                  color: activeTab === s.key ? s.color : "#64748b",
                  cursor: "pointer", fontSize: 13, whiteSpace: "nowrap", fontFamily: "inherit", fontWeight: activeTab === s.key ? 700 : 500,
                }}>
                  {s.icon} {s.label}
                  <span style={{ marginLeft: 5, padding: "1px 7px", borderRadius: 10, fontSize: 11, background: activeTab === s.key ? s.color : "#f1f5f9", color: activeTab === s.key ? "#fff" : "#94a3b8", fontWeight: 700 }}>
                    {results[s.key].length}
                  </span>
                </button>
              ))}
            </div>

            <div style={{ padding: "22px 26px", minHeight: 180 }}>
              <div style={{ display: "flex", alignItems: "center", gap: 12, marginBottom: 10 }}>
                <div style={{ width: 40, height: 40, borderRadius: 12, background: `${activeSection.color}18`, display: "flex", alignItems: "center", justifyContent: "center", fontSize: 20, border: `2px solid ${activeSection.color}33` }}>
                  {activeSection.icon}
                </div>
                <div style={{ flex: 1 }}>
                  <h3 style={{ margin: 0, fontSize: 16, fontWeight: 700, color: "#1e293b" }}>{activeSection.label}</h3>
                  <p style={{ margin: 0, fontSize: 12, color: "#94a3b8" }}>{filteredLines.length} item(s)</p>
                </div>
                <div style={{ fontSize: 11, color: "#94a3b8", fontStyle: "italic", textAlign: "right", lineHeight: 1.5 }}>💡 Line pe click karo<br/>AI + 🔊 voice milega</div>
              </div>

              {filteredLines.length > 0 ? (
                <div style={{ display: "flex", flexDirection: "column", gap: 8 }}>
                  {filteredLines.map((line, i) => (
                    <div key={i} onClick={() => { setSelectedTerm(cleanLine(line)); setShowAI(true); }} title="Click → AI explain + voice"
                      style={{ background: activeTab === "warnings" ? "#fffbeb" : `${activeSection.color}09`, border: `1px solid ${activeSection.color}28`, borderLeft: `4px solid ${activeSection.color}`, borderRadius: 10, padding: "10px 15px", fontSize: 14, color: "#334155", lineHeight: 1.75, cursor: "pointer", display: "flex", justifyContent: "space-between", alignItems: "center", gap: 10 }}
                      onMouseEnter={e => { e.currentTarget.style.background = `${activeSection.color}18`; e.currentTarget.style.boxShadow = `0 2px 10px ${activeSection.color}22`; }}
                      onMouseLeave={e => { e.currentTarget.style.background = activeTab === "warnings" ? "#fffbeb" : `${activeSection.color}09`; e.currentTarget.style.boxShadow = "none"; }}>
                      <span>{line.startsWith("-") ? "• " + cleanLine(line) : line}</span>
                      <span style={{ fontSize: 17, flexShrink: 0, opacity: 0.4 }}>🔊</span>
                    </div>
                  ))}
                </div>
              ) : (
                <div style={{ textAlign: "center", padding: "40px 20px", color: "#94a3b8" }}>
                  <div style={{ fontSize: 32, marginBottom: 8 }}>🔍</div>
                  <p style={{ margin: 0, fontSize: 14 }}>No information found for this section.</p>
                </div>
              )}
            </div>

            <div style={{ padding: "11px 22px", background: "#f8fafc", borderTop: "1px solid #e2e8f0", display: "flex", gap: 12, flexWrap: "wrap", alignItems: "center" }}>
              {SECTIONS.map(s => (
                <span key={s.key} style={{ fontSize: 12, color: results[s.key].length > 0 ? s.color : "#cbd5e1", fontWeight: results[s.key].length > 0 ? 600 : 400 }}>
                  {s.icon} {results[s.key].length}
                </span>
              ))}
              <span style={{ marginLeft: "auto", fontSize: 11, color: "#94a3b8", fontStyle: "italic" }}>Always consult your doctor</span>
            </div>
          </div>
        )}
      </div>

      <style>{`
        * { box-sizing: border-box; }
        button { transition: opacity 0.15s, transform 0.1s; }
        button:hover { opacity: 0.87; }
        button:active { transform: scale(0.97); }
        @keyframes spin { from { transform: rotate(0deg); } to { transform: rotate(360deg); } }
        ::-webkit-scrollbar { height: 5px; width: 5px; }
        ::-webkit-scrollbar-thumb { background: #cbd5e1; border-radius: 4px; }
        textarea::placeholder, input::placeholder { color: #94a3b8; }
      `}</style>
    </div>
  );
}

    </div>
  );
}
